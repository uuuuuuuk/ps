fun main() {
    val n = readln().toInt()

    for (i in 1..n) {
        val spaces = n - i
        val stars = 2 * i - 1
        println(" ".repeat(spaces) + "*".repeat(stars))
    }
    
    for (i in n-1 downTo 1) {
        val spaces = n - i
        val stars = 2 * i - 1
        println(" ".repeat(spaces) + "*".repeat(stars))
    }
}
